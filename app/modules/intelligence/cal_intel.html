/** * ============================================ * CAL_Foresight.gs - ADKINTOR FORESIGHT v1.0.0 * ============================================ * DESCRIPCIÓN: Inteligencia de calibración. * Dashboard de instrumentos, auditoría, * lista de vencidos/próximos, * tendencias de desviación, * exportación a PDF. * AUTOR: Mauricio * VERSIÓN: 1.0.1 * ÚLTIMA MODIFICACIÓN: 2026-07-16 * * 🚀 OPTIMIZACIÓN: * - Tab 1: Solo resumen agregado (no todo el historial) * - Tab 2: Filtrado en backend * - Tab 3: Bajo demanda (solo un instrumento) * ============================================ */ // ============================================ // CONFIGURACIÓN - LEE DESDE SETUP_CAL // ============================================ function getCALConfig() { const cache = CacheService.getScriptCache(); const cached = cache.get('CAL_CONFIG'); if (cached) return JSON.parse(cached); const configData = getSheetDataAsObjects('SETUP_CAL'); if (!configData || configData.length === 0) { return { alertDaysUrgent: 7, alertDaysUpcoming: 30, driftingThresholdPercent: 70, criticalDeviationPercent: 95 }; } const config = {}; configData.forEach(row => { const key = row.KEY; const value = row.VALUE; if (key === 'ALERT_DAYS_URGENT') config.alertDaysUrgent = parseFloat(value) || 7; if (key === 'ALERT_DAYS_UPCOMING') config.alertDaysUpcoming = parseFloat(value) || 30; if (key === 'DRIFTING_THRESHOLD_PERCENT') config.driftingThresholdPercent = parseFloat(value) || 70; if (key === 'CRITICAL_DEVIATION_PERCENT') config.criticalDeviationPercent = parseFloat(value) || 95; }); cache.put('CAL_CONFIG', JSON.stringify(config), 300); return config; } // ============================================ // TAB 1: DASHBOARD & AUDIT (SOLO RESUMEN) // ============================================ function CAL_getDashboardData() { try { // ✅ Solo cargar INSTRUMENT_MASTER (pequeña) const instruments = getLegacySheetWithCache('INSTRUMENT_MASTER'); if (!instruments || instruments.length === 0) { return { success: false, error: 'No instruments found' }; } // ✅ Para el resumen, solo necesitamos saber el ÚLTIMO resultado de cada Tag ID // NO cargamos todo el historial, solo los últimos resultados const historySummary = getLastResultsFromHistory(); const config = getCALConfig(); const urgentThreshold = config.alertDaysUrgent || 7; const today = new Date(); let overdue = 0; let urgent = 0; let failed = 0; const failedList = []; for (const instr of instruments) { const tagId = instr['Tag ID']; if (!tagId) continue; const name = instr['Instrument Name'] || tagId; const nextCalDate = instr['Next Calibration']; const lastResult = historySummary[tagId] || null; let daysToExpiry = null; if (nextCalDate) { const nextCal = new Date(nextCalDate); daysToExpiry = Math.ceil((nextCal - today) / (1000 * 60 * 60 * 24)); } if (daysToExpiry !== null && daysToExpiry < 0) { overdue++; } else if (daysToExpiry !== null && daysToExpiry <= urgentThreshold) { urgent++; } if (lastResult === 'Out of Tolerance' || lastResult === 'Fail') { failed++; failedList.push({ tagId, name, lastResult }); } } // Audit summary let compliant = 0; let nonCompliant = 0; let certificates = 0; for (const instr of instruments) { const nextCal = instr['Next Calibration']; if (nextCal) { const nextDate = new Date(nextCal); if (nextDate > today) { compliant++; } else { nonCompliant++; } } else { nonCompliant++; } if (instr['Certificate #']) certificates++; } // ✅ RECOMENDACIONES ESTRUCTURADAS const recommendations = []; // 1. Instrumentos vencidos if (nonCompliant > 0) { const overdueList = []; for (const instr of instruments) { const nextCal = instr['Next Calibration']; const tagId = instr['Tag ID'] || 'Unknown'; const name = instr['Instrument Name'] || tagId; if (nextCal) { const nextDate = new Date(nextCal); if (nextDate <= today) { // ✅ Formatear fecha: DD/MM/YYYY const day = String(nextDate.getDate()).padStart(2, '0'); const month = String(nextDate.getMonth() + 1).padStart(2, '0'); const year = nextDate.getFullYear(); const formattedDate = day + '/' + month + '/' + year; overdueList.push(`${tagId} - ${name} (due: ${formattedDate})`); } } else { overdueList.push(`${tagId} - ${name} (no calibration date set)`); } } if (overdueList.length > 0) { recommendations.push(`🔴 ${overdueList.length} instrument(s) are overdue. Calibrate immediately:`); for (const item of overdueList.slice(0, 10)) { recommendations.push(` • ${item}`); } if (overdueList.length > 10) { recommendations.push(` • ... and ${overdueList.length - 10} more`); } } } // 2. Certificados faltantes if (certificates < instruments.length) { const missingCertList = []; for (const instr of instruments) { if (!instr['Certificate #']) { const tagId = instr['Tag ID'] || 'Unknown'; const name = instr['Instrument Name'] || tagId; missingCertList.push(`${tagId} - ${name}`); } } if (missingCertList.length > 0) { recommendations.push(`📄 ${missingCertList.length} certificate(s) missing:`); for (const item of missingCertList.slice(0, 10)) { recommendations.push(` • ${item}`); } if (missingCertList.length > 10) { recommendations.push(` • ... and ${missingCertList.length - 10} more`); } } } // 3. Todo en orden if (nonCompliant === 0 && certificates === instruments.length) { recommendations.push('✅ All instruments are compliant and have certificates. Ready for audit.'); } // 4. Resumen adicional if (instruments.length === 0) { recommendations.push('📊 No instruments found in INSTRUMENT_MASTER.'); } return { success: true, summary: { total: instruments.length, overdue: overdue, urgent: urgent, failed: failed }, failedAlerts: failedList.slice(0, 10), audit: { total: instruments.length, compliant: compliant, nonCompliant: nonCompliant, certificates: certificates, recommendations: recommendations, auditReady: nonCompliant === 0 }, lastUpdated: new Date().toISOString() }; } catch (error) { logError('CAL', 'CAL_getDashboardData', error.message); console.error('Error:', error); return { success: false, error: error.toString() }; } } /** * ✅ NUEVA FUNCIÓN: Obtiene solo el último resultado de cada instrumento * SIN cargar todo el historial completo */ function getLastResultsFromHistory() { try { const history = getLegacySheetWithCache('CALIBRATION_HISTORY'); const result = {}; // ✅ Recorrer todo y siempre sobrescribir (así queda el último) for (const record of history) { const tagId = record['Tag ID']; if (!tagId) continue; // Siempre sobrescribir, así queda el último registro encontrado result[tagId] = record['Result']; } return result; } catch (error) { console.error('Error getting last results:', error); return {}; } } // ============================================ // TAB 2: OVERDUE & DUE (≤30 DAYS) // ============================================ function CAL_getOverdueAndDueList() { try { const instruments = getLegacySheetWithCache('INSTRUMENT_MASTER'); const assets = getLegacySheetWithCache('ASSET_REGISTER'); const config = getCALConfig(); const upcomingThreshold = config.alertDaysUpcoming || 30; const assetMap = {}; assets.forEach(asset => { const assetId = asset['Asset ID']; if (assetId) { assetMap[assetId] = asset['Equipment Name'] || assetId; } }); const today = new Date(); const results = []; for (const instr of instruments) { const tagId = instr['Tag ID']; if (!tagId) continue; const nextCalDate = instr['Next Calibration']; if (!nextCalDate) continue; const nextCal = new Date(nextCalDate); const daysToExpiry = Math.ceil((nextCal - today) / (1000 * 60 * 60 * 24)); if (daysToExpiry <= upcomingThreshold) { const assetId = instr['AssociatedAssetID']; results.push({ tagId: tagId, name: instr['Instrument Name'] || tagId, nextCalibration: nextCalDate, daysToExpiry: daysToExpiry, assetName: assetId ? assetMap[assetId] || null : null }); } } results.sort((a, b) => a.daysToExpiry - b.daysToExpiry); return results; } catch (error) { logError('CAL', 'getOverdueAndDueList', error.message); console.error('Error:', error); return []; } } // ============================================ // TAB 3: DRIFTING TRENDS (≥2 CALIBRATIONS) // ============================================ function CAL_getInstrumentsForTrend() { try { const instruments = getLegacySheetWithCache('INSTRUMENT_MASTER'); const history = getLegacySheetWithCache('CALIBRATION_HISTORY'); // ✅ Contar calibraciones por instrumento SIN cargar todo const calCount = {}; for (const record of history) { const tagId = record['Tag ID']; if (tagId) { calCount[tagId] = (calCount[tagId] || 0) + 1; } } const result = []; for (const instr of instruments) { const tagId = instr['Tag ID']; if (!tagId) continue; const count = calCount[tagId] || 0; if (count >= 2) { result.push({ tagId: tagId, name: instr['Instrument Name'] || tagId, historyCount: count }); } } return result.sort((a, b) => a.name.localeCompare(b.name)); } catch (error) { logError('CAL', 'getInstrumentsForTrend', error.message); console.error('Error:', error); return []; } } function CAL_getCALTrendData(tagId) { try { // ✅ Solo cargar historial de UN instrumento const history = getLegacySheetWithCache('CALIBRATION_HISTORY'); const instruments = getLegacySheetWithCache('INSTRUMENT_MASTER'); const instr = instruments.find(i => i['Tag ID'] === tagId); if (!instr) return { error: 'Instrument not found' }; // Filtrar solo este instrumento const instrHistory = history .filter(h => h['Tag ID'] === tagId) .sort((a, b) => new Date(a['Calibration Date']) - new Date(b['Calibration Date'])); // Extraer desviaciones const deviations = instrHistory .filter(h => { const hasDeviation = h['Deviation'] && !isNaN(parseFloat(h['Deviation'])); const hasMeasured = h['Measured Value'] && !isNaN(parseFloat(h['Measured Value'])); const hasExpected = h['Expected Value'] && !isNaN(parseFloat(h['Expected Value'])); return hasDeviation || (hasMeasured && hasExpected); }) .map(h => { let deviation = h['Deviation']; if ((deviation === undefined || deviation === null || isNaN(parseFloat(deviation))) && h['Measured Value'] && h['Expected Value']) { deviation = parseFloat(h['Measured Value']) - parseFloat(h['Expected Value']); } return { date: h['Calibration Date'], deviation: deviation !== null && !isNaN(parseFloat(deviation)) ? parseFloat(deviation) : null, measuredValue: h['Measured Value'] || null, expectedValue: h['Expected Value'] || null, result: h['Result'] || 'N/A', tolerance: h['Tolerance'] || null }; }) .filter(d => d.deviation !== null); const toleranceRaw = instr['Tolerance/Range'] || ''; const toleranceMatch = String(toleranceRaw).match(/([0-9.]+)/); const tolerance = toleranceMatch ? parseFloat(toleranceMatch[1]) : null; const config = getCALConfig(); const driftingThreshold = config.driftingThresholdPercent || 70; const criticalThreshold = config.criticalDeviationPercent || 95; let trend = 'insufficient'; let suggestion = ''; if (deviations.length >= 2 && tolerance && tolerance > 0) { let increasing = true; for (let i = 1; i < deviations.length; i++) { if (Math.abs(deviations[i].deviation) <= Math.abs(deviations[i-1].deviation)) { increasing = false; break; } } const lastValue = Math.abs(deviations[deviations.length-1].deviation); const percentOfTolerance = (lastValue / tolerance) * 100; if (increasing && percentOfTolerance > driftingThreshold) { trend = 'drifting'; suggestion = '⚠️ DRIFTING TREND DETECTED. Consider increasing calibration frequency.'; } else if (percentOfTolerance > criticalThreshold) { trend = 'critical'; suggestion = '🔴 CRITICAL: Deviation near tolerance limit. Immediate recalibration recommended.'; } else { trend = 'stable'; suggestion = '✅ Stable trend. No action needed.'; } } else if (deviations.length >= 2 && (!tolerance || tolerance === 0)) { trend = 'no-tolerance'; suggestion = '⚠️ No tolerance value configured. Cannot calculate trend percentage.'; } else if (deviations.length < 2) { trend = 'insufficient'; suggestion = '📊 Need at least 2 calibrations with deviation values to detect trend.'; } return { tagId: tagId, name: instr['Instrument Name'] || tagId, tolerance: toleranceRaw || 'Not configured', frequency: instr['Frequency (Months)'] || 12, deviations: deviations, trend: trend, suggestion: suggestion, historyCount: instrHistory.length, deviationCount: deviations.length }; } catch (error) { logError('CAL', 'getCALTrendData', error.message); console.error('Error in getCALTrendData:', error); return { error: error.toString() }; } } // ============================================ // EXPORT PDF // ============================================ function CAL_generateAuditPDF() { const dashboard = getCALDashboardData(); const list = CAL_getOverdueAndDueList(); const today = new Date(); if (!dashboard.success) { return HtmlService.createHtmlOutput('
