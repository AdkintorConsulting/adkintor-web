async function init() { loadUser(); showLoading('Loading KPI Dashboard...'); loadApiUrl(); if (!EAMS_API_URL) { hideLoading(); showToast('API URL not configured. Please login.', 'error'); return; } try { // ✅ Configurar mes/año actual var today = new Date(); var currentMonth = today.getMonth() + 1; var currentYear = today.getFullYear(); var monthSelect = document.getElementById('monthSelect'); var yearSelect = document.getElementById('yearSelect'); if (monthSelect) monthSelect.value = currentMonth; if (yearSelect) yearSelect.value = currentYear; // ✅ Configurar event listeners (tabs, botones) setupEventListeners(); // ✅ Pedir logo y versión al padre if (window.parent !== window) { window.parent.postMessage({ type: 'REQUEST_LOGO' }, '*'); window.parent.postMessage({ type: 'REQUEST_VERSION' }, '*'); } // ✅ Cargar datos await loadData(); hideLoading(); showToast('KPI Dashboard loaded successfully ✅', 'success'); } catch(err) { hideLoading(); showToast('Error loading data: ' + err.message, 'error'); } }
